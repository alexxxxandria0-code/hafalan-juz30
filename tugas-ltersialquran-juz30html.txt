        // Add color to Arabic text based on harakat
        function addColorToArabicText(text) {
            return text
                .replace(/Ÿé/g, '<span class="fathah">Ÿé</span>')
                .replace(/Ÿê/g, '<span class="kasrah">Ÿê</span>')
                .replace(/Ÿè/g, '<span class="dhammah">Ÿè</span>')
                .replace(/Ÿí/g, '<span class="sukun">Ÿí</span>')
                .replace(/Ÿã|Ÿç|Ÿå/g, '<span class="tanwin">$&</span>')
                .replace(/Ÿë/g, '<span class="tasydid">Ÿë</span>');
        }

        // Setup event listeners for ayat interactions
        function setupAyatEventListeners() {
            // Play button for each ayat
            document.querySelectorAll('.play-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const ayatNumber = parseInt(e.currentTarget.dataset.ayat);
                    playAyat(ayatNumber);
                });
            });
            
            // Repeat button for each ayat
            document.querySelectorAll('.repeat-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const ayatNumber = parseInt(e.currentTarget.dataset.ayat);
                    toggleRepeatAyat(ayatNumber);
                });
            });
            
            // Tafsir expand/collapse
            document.querySelectorAll('.tafsir-header').forEach(header => {
                header.addEventListener('click', () => {
                    const content = header.nextElementSibling;
                    content.classList.toggle('expanded');
                    const icon = header.querySelector('i');
                    icon.classList.toggle('fa-chevron-down');
                    icon.classList.toggle('fa-chevron-up');
                });
            });
            
            // Memorization checkbox
            document.querySelectorAll('.checkbox').forEach(checkbox => {
                checkbox.addEventListener('click', (e) => {
                    const surah = parseInt(e.currentTarget.dataset.surah);
                    const ayat = parseInt(e.currentTarget.dataset.ayat);
                    toggleMemorized(surah, ayat);
                });
            });
            
            // Bookmark button
            document.querySelectorAll('.bookmark-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const surah = parseInt(e.currentTarget.dataset.surah);
                    const ayat = parseInt(e.currentTarget.dataset.ayat);
                    toggleBookmark(surah, ayat);
                });
            });
            
            // Double click to play audio
            document.querySelectorAll('.ayat-card').forEach(card => {
                card.addEventListener('dblclick', () => {
                    const ayatNumber = parseInt(card.querySelector('.ayat-number').textContent);
                    playAyat(ayatNumber);
                });
            });
            
            // Single click to highlight
            document.querySelectorAll('.ayat-card').forEach(card => {
                card.addEventListener('click', (e) => {
                    if (!e.target.closest('.ayat-controls') && 
                        !e.target.closest('.hafalan-controls') &&

                        updateAudioInfo();
                    }
                });
            });
        }

                btn.innerHTML = '<i class="fas fa-play"></i>';
            });
        }

        function prevAyat() {
            if (state.currentAyat > 1) {
                playAyat(state.currentAyat - 1);
            } else if (state.currentSurah > 78) {
                // Go to previous surah
                const prevSurah = state.currentSurah - 1;
                loadSurah(prevSurah);
                const prevSurahData = surahList.find(s => s.number === prevSurah);
                playAyat(prevSurahData.verses);
            }
        }

        function nextAyat() {
            const currentSurahData = surahList.find(s => s.number === state.currentSurah);
            
            if (state.currentAyat < currentSurahData.verses) {
                playAyat(state.currentAyat + 1);
            } else if (state.currentSurah < 114) {
                // Go to next surah
                const nextSurah = state.currentSurah + 1;
                loadSurah(nextSurah);
                playAyat(1);
                
                // Show notification if auto-play is on
                if (state.autoPlay) {
                    showNotification(`Beralih ke surah ${surahList.find(s => s.number === nextSurah).name}`);
                }
            } else {
                // End of Juz 30
                showNotification("Selamat! Anda telah menyelesaikan Juz 30");
                pauseAudio();
            }
        }

        // Progress tracking functions
        function toggleMemorized(surah, ayat) {
            const key = `${surah}-${ayat}`;
            
            if (state.progress[key]) {
                delete state.progress[key];
            } else {
                state.progress[key] = true;
            }
            
            // Update UI
            const checkbox = document.querySelector(`.checkbox[data-surah="${surah}"][data-ayat="${ayat}"]`);
            checkbox.classList.toggle('checked');
            
            // Save to localStorage
            localStorage.setItem('hafalanProgress', JSON.stringify(state.progress));
            
            // Update progress display
            updateProgressDisplay();
            renderSurahList();
            
            // Show notification
            showNotification("Progress hafalan telah disimpan!");
        }

        function toggleBookmark(surah, ayat) {
            const key = `${surah}-${ayat}`;
            const index = state.bookmarks.indexOf(key);
            
            if (index > -1) {
                state.bookmarks.splice(index, 1);
            } else {
                state.bookmarks.push(key);
            }
            
            // Update UI
            const bookmarkBtn = document.querySelector(`.bookmark-btn[data-surah="${surah}"][data-ayat="${ayat}"]`);
            bookmarkBtn.classList.toggle('active');
            
            // Save to localStorage
            localStorage.setItem('bookmarks', JSON.stringify(state.bookmarks));
        }

        function isAyatMemorized(surah, ayat) {
            return !!state.progress[`${surah}-${ayat}`];
        }

        function isAyatBookmarked(surah, ayat) {
            return state.bookmarks.includes(`${surah}-${ayat}`);
        }

        function calculateSurahProgress(surahNumber) {
            const surah = surahList.find(s => s.number === surahNumber);
            let memorizedCount = 0;
            
            for (let i = 1; i <= surah.verses; i++) {
                if (isAyatMemorized(surahNumber, i)) {
                    memorizedCount++;
                }
            }
            
            return Math.round((memorizedCount / surah.verses) * 100);
        }

        function calculateOverallProgress() {
            let totalAyat = 0;
            let memorizedAyat = 0;
            
            surahList.forEach(surah => {
                totalAyat += surah.verses;
                
                for (let i = 1; i <= surah.verses; i++) {
                    if (isAyatMemorized(surah.number, i)) {
                        memorizedAyat++;
                    }
                }
            });
            
            return Math.round((memorizedAyat / totalAyat) * 100);
        }

        function updateProgressDisplay() {
            const progress = calculateOverallProgress();
            elements.progressValue.textContent = `${progress}%`;
            elements.progressCircle.style.background = `conic-gradient(var(--primary-light) ${progress}%, var(--border-color) 0%)`;
        }

        // Utility functions
        function formatTime(seconds) {
            const mins = Math.floor(seconds / 60);
            const secs = Math.floor(seconds % 60);
            return `${mins}:${secs < 10 ? '0' : ''}${secs}`;
        }

        function handleKeyboardShortcuts(e) {
            // Prevent shortcuts when typing in search box
            if (e.target.tagName === 'INPUT') return;
            
            switch(e.key) {
                case ' ':
                    e.preventDefault();
                    togglePlayPause();
                    break;
                case 'ArrowRight':
                    e.preventDefault();
                    nextAyat();
                    break;
                case 'ArrowLeft':
                    e.preventDefault();
                    prevAyat();
                    break;
                case 'r':
                case 'R':
                    e.preventDefault();
                    toggleRepeat();
                    break;
            }
        }

            let currentTime = 0;
            const duration = 30; // Mock duration in seconds

            elements.durationEl.textContent = formatTime(duration);
            elements.progressBarFill.style.width = '0%';

                elements.currentTimeEl.textContent = formatTime(currentTime);
                elements.progressBarFill.style.width = `${(currentTime / duration) * 100}%`;
                
                if (currentTime >= duration) {
                    clearInterval(interval);
                    
                    if (state.repeatMode) {
                        // Repeat current ayat
                        playAyat(state.currentAyat);
                    } else if (state.autoPlay) {
                        // Auto-play next ayat
                        nextAyat();
                    } else {
                        // Stop playback
                        pauseAudio();
                    }
                }
            }, 100);
        }
